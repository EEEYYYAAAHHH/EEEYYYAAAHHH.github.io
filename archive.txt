// script.js
//use the old one for reference...

//wait until the page fully loads
//helps avoid the page from not loading at all
document.addEventListener("DOMContentLoaded", () => {
  const listEl = document.getElementById("char-list");
  const portraitEl = document.getElementById("p-portrait");
  const nameEl = document.getElementById("p-name");
  const worldEl = document.getElementById("p-world");
  const statsEl = document.getElementById("p-stats");
  const moveEl = document.getElementById("p-moveset"); 
  const bioEl = document.getElementById("p-bio");
  const skillsListEl = document.getElementById("skill-list");

  //load the JSON file
  //refer to CPP on how to load a file
  fetch("gallery.json")
    .then(response => response.json())
    .then(data => {
      listEl.innerHTML = ""; // clear placeholder

      //loop through all characters and make list items
	  //parsing maybe? 
	  //its already parsed
      data.forEach((character) => {
        const li = document.createElement("li");
        li.textContent = character.name;
        li.classList.add("char-item");
        li.addEventListener("click", () => showCharacter(character));
        listEl.appendChild(li);
      });
    })
	// need a error catch 
    .catch(err => {
      console.error("Error loading gallery.json:", err);
      listEl.innerHTML = "<li class='error'>Failed to load character data.</li>";
    });

////////////////////////////////////////////////////////////////////////

  //need a function to display a characterâ€™s details
  //might need AI to help witht his part
  function showCharacter(character) {
    portraitEl.src = character.portrait;
    //portraitEl.alt = `${character.name} portrait`;
    nameEl.textContent = character.name;
    worldEl.textContent = `World: ${character.world}`;

    //fill in stats (power, fortitude, etc.)
    statsEl.innerHTML = "";
    for (const [stat, value] of Object.entries(character)) {
      const statNames = ["power", "fortitude", "tactical acumen", "fighting skill", "range", "arts"];
      if (statNames.includes(stat)) {
        const li = document.createElement("li");
        li.textContent = `${stat.toUpperCase()}: ${value}`;
        statsEl.appendChild(li);
      }
    }

    // Fill in moveset (array of strings in your JSON)
    moveEl.innerHTML = "";
    if (Array.isArray(character.moveset)) {
      character.moveset.forEach(move => {
        const li = document.createElement("li");
        li.textContent = move;
        moveEl.appendChild(li);
      });
    } else {
      const li = document.createElement("li");
      li.textContent = "No moves listed.";
      moveEl.appendChild(li);
    }

    // Fill in bio
    bioEl.textContent = character.bio;

    // Fill in skills
    skillsListEl.innerHTML = "";
    if (Array.isArray(character.skills)) {
      character.skills.forEach(skill => {
        const li = document.createElement("li");
        li.textContent = skill;
        skillsListEl.appendChild(li);
      });
    }
  }
});
